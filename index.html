    <script>
        function nav(n) {
            document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
            document.getElementById('page' + n).classList.add('active');
            if(n===2) initWYR();
            if(n===3) initColors();
            if(n===4) initQuiz();
        }

        let wyrIndex = 0;
        const wyrQuestions = Array.from({length: 100}, (_, i) => ({
            q: `ئەگەر ناچار بیت، کامیان هەڵدەبژێریت؟ (${i+1})`,
            o1: `بژاردەی یەکەم`,
            o2: `بژاردەی دووەم`,
            p: Math.floor(Math.random() * 80) + 10
        }));

        function initWYR() {
            const data = wyrQuestions[wyrIndex];
            document.getElementById('wyr-count').innerText = `پرسیاری ${wyrIndex+1} لە ١٠٠`;
            document.getElementById('wyr-q').innerText = data.q;
            document.getElementById('opt1').innerText = data.o1;
            document.getElementById('opt2').innerText = data.o2;
            document.getElementById('bar1').style.width = '0%';
            document.getElementById('bar2').style.width = '0%';
            document.getElementById('wyr-perc').classList.add('hidden');
        }

        function vote(c) {
            const data = wyrQuestions[wyrIndex];
            document.getElementById('bar1').style.width = data.p + '%';
            document.getElementById('bar2').style.width = (100 - data.p) + '%';
            document.getElementById('wyr-perc').innerText = `%${data.p} - %${100-data.p}`;
            document.getElementById('wyr-perc').classList.remove('hidden');
            setTimeout(() => {
                wyrIndex++;
                if(wyrIndex < 100) initWYR(); else nav(3);
            }, 1500);
        }

        const colors = [
            {n: 'سوور', c: '#ef4444', m: 'تێکەڵەی ماجێنتا و زەرد'},
            {n: 'شین', c: '#3b82f6', m: 'ڕەنگی سەرەکی'},
            {n: 'سەوز', c: '#22c55e', m: 'تێکەڵەی شین و زەرد'},
            {n: 'زەرد', c: '#eab308', m: 'ڕەنگی سەرەکی'},
            {n: 'مۆر', c: '#a855f7', m: 'تێکەڵەی سوور و شین'}
        ];

        function initColors() {
            const grid = document.getElementById('color-grid');
            if(grid.innerHTML !== "") return;
            colors.forEach(col => {
                const btn = document.createElement('div');
                btn.className = "glass h-20 cursor-pointer transition-transform hover:scale-110";
                btn.style.backgroundColor = col.c;
                btn.onclick = () => {
                    document.getElementById('main-color-display').style.backgroundColor = col.c;
                    document.getElementById('color-info').innerText = `${col.n}: ${col.m}`;
                };
                grid.appendChild(btn);
            });
        }

        let quizIdx = 0;
        const quizData = [
            {q: "پایتەختی فەڕەنسا کوێیە؟", a: "پاریس"},
            {q: "گەورەترین گیانلەبەری سەر زەوی؟", a: "فیل"},
            {q: "٢٥ + ٢٥ دەکاتە چەند؟", a: "٥٠"}
        ];

        function initQuiz() {
            if(quizIdx >= quizData.length) { nav(5); return; }
            document.getElementById('quiz-count').innerText = `${quizIdx+1} / ${quizData.length}`;
            document.getElementById('quiz-q').innerText = quizData[quizIdx].q;
            document.getElementById('quiz-input').value = '';
            document.getElementById('quiz-feedback').innerText = '';
        }

        function checkQuiz() {
            const userAns = document.getElementById('quiz-input').value.trim();
            if(userAns === quizData[quizIdx].a) {
                document.getElementById('quiz-feedback').innerText = "ڕاستە! 🎉";
                setTimeout(() => { quizIdx++; initQuiz(); }, 1000);
            } else {
                document.getElementById('quiz-feedback').innerText = "هەڵەیە ❌";
            }
        }

        function postComment() {
            const txt = document.getElementById('comment-text').value;
            if(!txt) return;
            const div = document.createElement('div');
            div.className = "glass p-4 text-lg border-r-4 border-blue-400 mb-2";
            div.innerText = txt;
            document.getElementById('comments-list').prepend(div);
            document.getElementById('comment-text').value = '';
        }
